조건문

키워드 설명 예제
if 조건문을 나타내는 키워드 if (x > 10) {
    alert('x is greater than 10');
} else {
    alert('x is less than or equal to 10');
}
if-else 조건에 따라 다른 동작을 수행하는 조건문 var age = 18;
if (age >= 18) {
    alert('성인입니다.');
} else {
    alert('미성년자입니다.');
}
switch 다중 조건문을 나타내는 키워드 var day = 'Monday';
switch (day) {
    case 'Monday':
        alert('Today is Monday.');
        break;
    case 'Tuesday':
        alert('Today is Tuesday.');
        break;
    default:
        alert('Today is not Monday or Tuesday.');
}

반복문

키워드 설명 예제
배열 여러 개의 요소를 저장하는 자료형 var array = [273, 'banana', true, function(){},{}];
alert(array[0]); // '273'
for 반복문을 나타내는 키워드 for (var i = 0; i < 5; i++) {
    alert(i);
}
while 조건이 참인 동안 반복하는 반복문 var i = 0;
while (i < 5) {
    alert(i);
    i++;
}
